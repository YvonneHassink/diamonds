# Get the current date and time
current_time <- Sys.time()

# Create the plot
p <- ggplot(diamonds, aes(x=carat, y=price)) +
  geom_point(alpha=0.5, color='blue') +
  labs(title='Diamonds Dataset: Carat vs Price',

       caption=paste('Generated on:', current_time)) +
  theme_minimal()
